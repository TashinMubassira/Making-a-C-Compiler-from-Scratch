PRINT PROC
	CMP AX, 0H
	JE  AX_IS_ZERO
	JL  AX_IS_NEGATIVE
	JMP AX_IS_NOT_NEGATIVE

	AX_IS_ZERO:
		MOV AH,2
		MOV DL,30H
		INT 21H
		JMP EXIT_PRINT
	
	AX_IS_NEGATIVE:
		NEG AX
		MOV BX,AX
		MOV AH, 2
		MOV DL, '-'
		INT 21H
		MOV AX,BX
	
	AX_IS_NOT_NEGATIVE:
	MOV CX, 0H
	MOV DX, 0H
	CHECK_NUMBER:
		CMP AX, 0H
		JE  PRINT_NUMBER
		
		MOV BX, 0AH
		IDIV BX				
		
		PUSH DX
		INC CX
		
		XOR DX, DX
		JMP CHECK_NUMBER
	
	PRINT_NUMBER:
		CMP CX, 0H
		JE EXIT_PRINT
		
		POP DX
		ADD DX, 30H
		MOV AH, 2
		INT 21H
		
		DEC CX
		JMP PRINT_NUMBER
		
	EXIT_PRINT:
		RET 
		PRINT ENDP
